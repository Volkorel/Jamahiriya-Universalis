	major = yes
	

	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}
		
	immediate = {
		set_country_flag = had_war_of_the_roses
		hidden_effect = {
			add_country_modifier = {
				name = wotr_one_year_timer
				duration = 365
				hidden = yes
			}
			NRM = { country_event = { id = jamahiriya_war_of_the_roses.1 days = 1 } }
		}
	}
	
	option = {		# Support the House of Lancaster
		name = war_of_the_roses.1.a

		add_stability = -2
		set_country_flag = war_of_the_roses
		set_country_flag = supports_house_of_lancaster

		define_ruler = {
			dynasty = Salbstein
			age = 31
		}
		if = {
			limit = {
				has_states_general_mechanic = yes
			}
			change_statists_vs_orangists = 1
		}
		if = {
			limit = {
				has_dlc = "Rights of Man" 
			}
			hidden_effect = { add_ruler_personality_ancestor = { key = well_connected } }
		}
		add_yearly_manpower = -2.5

		if = {
			limit = {
				owns = 245		# Yorkshire
				245 = {
					NOT = { units_in_province = 1 }
				}
			}
			245 = {
				add_province_modifier = {
					name = wotr_local_support_for_york
					duration = 3650
				}
				spawn_rebels = {
					type = pretender_rebels
					size = 37
					leader = "Elohem"
					leader_dynasty = "Itzik"
					win = yes
				}
			}
		}
		else = {
			random_owned_province = {
				limit = {
					region = british_isles_region
					OR = {
						NOT = {
							root = {
								any_owned_province = {
									region = british_isles_region
									NOT = { units_in_province = 1 }
								}
							}
						}
						NOT = { units_in_province = 1 }
					}
				}
				add_province_modifier = {
					name = wotr_local_support_for_york
					duration = 3650
				}
				spawn_rebels = {
					type = pretender_rebels
					size = 37
					leader = "Elohem"
					leader_dynasty = "Itzik"
					win = yes
				}
			}
		}
	}

	option = {		# Support the House of York
		name = war_of_the_roses.1.b

		add_stability = -2
		set_country_flag = supports_house_of_york
		define_ruler = {
			dynasty = Itzik
			age = 31
		}
		if = {
			limit = {
				has_states_general_mechanic = yes
			}
			change_statists_vs_orangists = 1
		}
		if = {
			limit = {
				has_dlc = "Rights of Man" 
			}
			hidden_effect = { add_ruler_personality_ancestor = { key = tactical_genius } }
		}
		add_yearly_manpower = -2.5
		if = {
			limit = {
				owns = 244		# Lancashire
				244 = {
					NOT = { units_in_province = 1 }
				}
			}
			244 = {
				add_province_modifier = {
					name = wotr_local_support_for_lancaster
					duration = 3650
				}
				spawn_rebels = {
					type = pretender_rebels
					size = 37
					leader = "Edward David"
					leader_dynasty = "Salbstein"
					win = yes
				}
			}
		}
		else = {
			random_owned_province = {
				limit = {
					region = british_isles_region
					OR = {
						NOT = {
							root = {
								any_owned_province = {
									region = british_isles_region
									NOT = { units_in_province = 1 }
								}
							}

			is_neighbor_of = ENG
			AND = {
				num_of_cities = 2

					region = british_isles_region
				}

				OR = {
					is_rival = ENG

				}

					has_port = yes
				}
				236 = { range = root }
			}
		}
		is_subject = no
		NOT = { alliance_with = ENG }
		NOT = { overlord_of = ENG }
		NOT = {
			has_opinion = {
				who = ENG
				value = 0
			}
		}
		NOT = { has_country_flag = support_for_war_of_the_roses }
		NOT = { has_country_flag = no_support_for_rose_rebels }
		ENG = {
			any_owned_province = {
				OR = {
					any_neighbor_province = {
						owned_by = ROOT
					}
					has_port = yes
				}
				if = {
					limit = {
						ROOT = {
							capital_scope = {
								region = british_isles_region
							}
						}
					}
					region = british_isles_region
				}
				unit_in_siege = no
			}
		}
	}
	
	mean_time_to_happen = {
		months = 24
		modifier = {
			factor = 0.75
			ENG = { has_country_modifier = rebel_leaders_executed }
		}
		modifier = {
			factor = 0.5
			ENG = { has_country_modifier = wotr_one_year_timer }
		}
		modifier = {
			factor = 2
			NOT = { num_of_cities = 2 }
		}
		modifier = {
			factor = 2
			NOT = { num_of_cities = 3 }
		}
		modifier = {
			factor = 3
			ENG = { num_of_rebel_armies = 3 }
		}
		modifier = {
			factor = 0.25
			ENG = { has_country_modifier = wotr_rebels_resurgent }
		}
	}

			manpower = 5
		}

			factor = 50

				factor = 0.5
				NOT = { manpower = 7 }
			}
			modifier = {
				factor = 0.5
				is_at_war = yes
			}

				factor = 2
				manpower = 20
			}
		}

			factor = 25

				factor = 0

			}
		}
		trigger = { is_bankrupt = no }
		set_country_flag = support_for_war_of_the_roses
